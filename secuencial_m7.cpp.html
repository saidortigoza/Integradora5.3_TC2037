#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main(int argc, char* argv[]){
int t;
cin >> t;

while (t--){
int n, k;
cin >> n >> k;
int a[n];
for (int i = 0; i < n; i++){
cin >> a[i];
}
sort(a,a + n);
int maxSpell;
maxSpell = 0;
for (int i = 1; i < n; i++){
maxSpell += (k - a[i]) / a[0];
}
cout << maxSpell << ;
}

return 0;
}
